create table categories (
    id int primary key auto_increment,
    name varchar(50) not null
);

create table manufacturers (
    id int primary key auto_increment,
    name varchar(50) not null
);

create table suppliers (
    id int primary key auto_increment,
    name varchar(50) not null
);

create table order_statuses (
    id int primary key auto_increment,
    name varchar(50) not null
);

create table users (
    id int primary key auto_increment,
    login varchar(50) not null unique,
    password varchar(50) not null,
    role varchar(50) not null
);

create table products (
    id int primary key auto_increment,
    article varchar(50) not null,
    name varchar(50) not null,
    description varchar(255) not null,
    price decimal(10,2) not null,
    unit varchar(50) not null,
    stock int not null,
    discount decimal(5,2) default 0,
    image_path varchar(255),
    category_id int,
    manufacturer_id int,
    supplier_id int,
    foreign key (category_id) references categories(id) on delete set null,
    foreign key (manufacturer_id) references manufacturers(id) on delete set null,
    foreign key (supplier_id) references suppliers(id) on delete set null
);

create table orders (
    id int primary key auto_increment,
    user_id int not null,
    status_id int not null,
    address varchar(255) not null,
    order_date date not null,
    delivery_date date not null,
    foreign key (user_id) references users(id),
    foreign key (status_id) references order_statuses(id)
);

create table order_items (
    order_id int,
    product_id int,
    quantity int not null default 1,
    primary key (order_id, product_id),
    foreign key (order_id) references orders(id) on delete cascade,
    foreign key (product_id) references products(id) on delete restrict
);

import pymysql

class Databasee:
    def __init__(self):
        self.conn = pymysql.connect(
            host="localhost",
            user="root",
            password="",
            db="shop1",
            cursorclass=pymysql.cursors.Cursor,
            charset="utf8mb4"
        )

    def select_all(self, query, params=()):
        cursor = self.conn.cursor()
        cursor.execute(query, params)
        rows = cursor.fetchall()
        cursor.close()
        return rows

    def execute(self, query, params=()):
        cursor = self.conn.cursor()
        cursor.execute(query, params)
        self.conn.commit()
        cursor.close()

def get_discount_number(text):
    # извлечение скидки: какое там число?

        return int(float(text))

        return 0

def get_or_create_id(table, name_value):
    # поиск айди: есть ли уже такая запись в справочнике?
    res = db.select_all(f"select id from {table} where name = %s", (name_value,))
    if res:
        return res[0][0]
    db.execute(f"insert into {table} (name) values (%s)", (name_value,))
    res = db.select_all(f"select id from {table} where name = %s", (name_value,))
    return res[0][0]


class LoginDialog(QDialog):

        self.resize(250, 250)

        self.login_edit = QLineEdit()

        login_btn = QPushButton("Войти")
        guest_btn = QPushButton("Гость")
        login_btn.clicked.connect(self.try_login)
        guest_btn.clicked.connect(self.login_guest)

    def try_login(self):
        # авторизация пользователя: подходят ли логин и новый пароль?
        login = self.login_edit.text().strip()
        password = self.password_edit.text().strip()

        if login == "admin" and password == "admin123":
            self.selected_user = "admin"
            self.selected_role = "Администратор"
            self.accept()
            return

        if login == "manager" and password == "manager123":
            self.selected_user = "manager"
            self.selected_role = "Менеджер"
            self.accept()
            return

        if login == "client" and password == "client123":
            self.selected_user = "client"
            self.selected_role = "Пользователь"
            self.accept()
            return

    def login_guest(self):
        # вход гостя: пускать ли без авторизации?
        self.selected_user = "guest"
        self.selected_role = "Гость"
        self.accept()

    def refresh_tab(self):
        index = self.tabs.currentIndex()
        self.tabs.clear()

        self.tabs.addTab(self.product_tab(), "Товары")

        if self.role in ["Администратор", "Менеджер"]:
            self.tabs.addTab(self.order_tab(), "Заказы")

        if index < self.tabs.count():
            self.tabs.setCurrentIndex(index)

    def clear_layout(self, layout):
        # очистка разметки: остались ли старые виджеты?
        while layout.count():
            layout.takeAt(0).widget().deleteLater()

    def product_tab(self):
        page = QWidget()
        layout = QVBoxLayout(page)

        top_layout = QHBoxLayout()

        self.search_edit = None
        self.sort_combo = None
        self.price_combo = None
        self.discount_combo = None
        self.supplier_combo = None

        if self.role in ["Администратор", "Менеджер"]:
            self.search_edit = QLineEdit()
            self.search_edit.setPlaceholderText("Поиск")
            self.search_edit.textChanged.connect(self.show_product)
            top_layout.addWidget(self.search_edit)

            self.sort_combo = QComboBox()
            self.sort_combo.addItem("Без сорт. количества")
            self.sort_combo.addItem("Количество по возрастанию")
            self.sort_combo.addItem("Количество по убыванию")
            top_layout.addWidget(self.sort_combo)

            self.price_combo = QComboBox()
            self.price_combo.addItem("Без сорт. цены")
            self.price_combo.addItem("Цена по возрастанию")
            self.price_combo.addItem("Цена по убыванию")
            top_layout.addWidget(self.price_combo)

            self.discount_combo = QComboBox()
            self.discount_combo.addItem("Без сорт. скидки")
            self.discount_combo.addItem("Скидка по возрастанию")
            self.discount_combo.addItem("Скидка по убыванию")
            top_layout.addWidget(self.discount_combo)

            # сброс конфликтов: как переключать режимы без багов?
            def handle_sort_change():
                sender = self.sender()
                if sender == self.sort_combo and self.sort_combo.currentIndex() != 0:
                    self.price_combo.blockSignals(True)
                    self.discount_combo.blockSignals(True)
                    self.price_combo.setCurrentIndex(0)
                    self.discount_combo.setCurrentIndex(0)
                    self.price_combo.blockSignals(False)
                    self.discount_combo.blockSignals(False)
                elif sender == self.price_combo and self.price_combo.currentIndex() != 0:
                    self.sort_combo.blockSignals(True)
                    self.discount_combo.blockSignals(True)
                    self.sort_combo.setCurrentIndex(0)
                    self.discount_combo.setCurrentIndex(0)
                    self.sort_combo.blockSignals(False)
                    self.discount_combo.blockSignals(False)
                elif sender == self.discount_combo and self.discount_combo.currentIndex() != 0:
                    self.sort_combo.blockSignals(True)
                    self.price_combo.blockSignals(True)
                    self.sort_combo.setCurrentIndex(0)
                    self.price_combo.setCurrentIndex(0)
                    self.sort_combo.blockSignals(False)
                    self.price_combo.blockSignals(False)
                self.show_product()

            self.sort_combo.currentTextChanged.connect(handle_sort_change)
            self.price_combo.currentTextChanged.connect(handle_sort_change)
            self.discount_combo.currentTextChanged.connect(handle_sort_change)

            self.supplier_combo = QComboBox()
            self.supplier_combo.addItem("Все поставщики")

            # сбор поставщиков: какие уникальные имена есть?
            suppliers = db.select_all("select name from suppliers order by name")
            for row in suppliers:
                self.supplier_combo.addItem(str(row[0]))

            self.supplier_combo.currentTextChanged.connect(self.show_product)
            top_layout.addWidget(self.supplier_combo)

            if self.role == "Администратор":
                add = QPushButton("Добавить товар")
                add.clicked.connect(self.add_product)
                top_layout.addWidget(add)

        layout.addLayout(top_layout)

        self.product_layout = QVBoxLayout()
        content = QWidget()
        content.setLayout(self.product_layout)

        scroll = QScrollArea()
        scroll.setWidgetResizable(True)
        scroll.setWidget(content)

        layout.addWidget(scroll)
        self.show_product()
        return page

    def get_product(self):
        # запрос с джоинами: связали ли мы справочники?
        sql = """
            select p.id, p.article, c.name, p.name, p.description, m.name, s.name,
                   p.price, p.unit, p.stock, p.discount, p.image_path
            from products p
            left join categories c on p.category_id = c.id
            left join manufacturers m on p.manufacturer_id = m.id
            left join suppliers s on p.supplier_id = s.id
        """
        rows = list(db.select_all(sql))

        if self.search_edit:
            text = self.search_edit.text().lower()
            if text != "":
                rows = [r for r in rows if text in str(r).lower()]

        if self.sort_combo and self.sort_combo.currentIndex() != 0:
            # сортировка по количеству: в какую сторону крутить карточки?
            s = self.sort_combo.currentText()
            if s == "Количество по возрастанию":
                rows.sort(key=lambda x: x[9])
            elif s == "Количество по убыванию":
                rows.sort(key=lambda x: x[9], reverse=True)

        if self.price_combo and self.price_combo.currentIndex() != 0:
            # сортировка по цене: дорогой или дешевый товар ищем?
            s = self.price_combo.currentText()
            if s == "Цена по возрастанию":
                rows.sort(key=lambda x: x[7])
            elif s == "Цена по убыванию":
                rows.sort(key=lambda x: x[7], reverse=True)

        if self.discount_combo and self.discount_combo.currentIndex() != 0:
            # сортировка по скидке: где у нас максимальная выгода?
            s = self.discount_combo.currentText()
            if s == "Скидка по возрастанию":
                rows.sort(key=lambda x: x[10])
            elif s == "Скидка по убыванию":
                rows.sort(key=lambda x: x[10], reverse=True)

        if self.supplier_combo:
            sup = self.supplier_combo.currentText()
            if sup != "Все поставщики":
                rows = [r for r in rows if r[6] == sup]

        return rows

    def show_product(self):

    def add_product(self):
        dialog = ProductDialog()
        if not dialog.exec():
            return

        data = dialog.get_data()
        cat_id = get_or_create_id("categories", data[1])
        man_id = get_or_create_id("manufacturers", data[4])
        sup_id = get_or_create_id("suppliers", data[5])
        disc_val = get_discount_number(data[9])

        # сохранение нового товара: занесли ли внешние ключи?
        sql = """
            insert into products (article, name, description, price, unit, stock, discount, image_path, category_id, manufacturer_id, supplier_id)
            values (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
        """
        db.execute(sql, (data[0], data[2], data[3], data[6], data[7], data[8], disc_val, data[10], cat_id, man_id, sup_id))
        self.set_status("Товар добавлен")
        self.refresh_tab()

    def edit_product(self, old_product):
        dialog = ProductDialog(old_product)
        if not dialog.exec():
            return

        data = dialog.get_data()
        cat_id = get_or_create_id("categories", data[1])
        man_id = get_or_create_id("manufacturers", data[4])
        sup_id = get_or_create_id("suppliers", data[5])
        disc_val = get_discount_number(data[9])

        sql = """
            update products
            set article=%s, name=%s, description=%s, price=%s, unit=%s, stock=%s, discount=%s, image_path=%s,
                category_id=%s, manufacturer_id=%s, supplier_id=%s
            where id=%s
        """

        # список параметров: тут тоже ровно 12 элементов?
        params = (
            data[0],  # 1. артикул
            data[2],  # 2. имя
            data[3],  # 3. описание
            data[6],  # 4. цена
            data[7],  # 5. ед изм
            data[8],  # 6. кол во
            disc_val,  # 7. скидка числом
            data[10],  # 8. фото путь
            cat_id,  # 9. айди категории
            man_id,  # 10. айди производителя
            sup_id,  # 11. айди поставщика
            old_product[0]  # 12. уникальный айди из where
        )

        db.execute(sql, params)
        self.set_status("Товар изменен")
        self.refresh_tab()

    def delete_product(self, product):
        # проверка состава: есть ли этот товар в заказах?
        in_order = db.select_all("select * from order_items where product_id=%s", (product[0],))
        if in_order:
            QMessageBox.warning(self, "Предупреждение", "Товар есть в заказах")
            return

        answer = QMessageBox.question(self, "Удаление", "Точно ли удалить товар")
        if answer == QMessageBox.StandardButton.Yes:
            db.execute("delete from products where id=%s", (product[0],))
            self.set_status("Товар удален")
            self.refresh_tab()

    def order_tab(self):

        scroll = QScrollArea()
        scroll.setWidgetResizable(True)

        sql = """
            select o.id, concat('Заказ №', o.id), s.name, o.address, o.order_date, o.delivery_date, u.login, 'Состав заказа'
            from orders o
            join users u on o.user_id = u.id
            join order_statuses s on o.status_id = s.id
            order by o.id desc

        orders = db.select_all(sql)

        for order in orders:
            card_layout.addWidget(self.order_card(order))

        card_layout.addStretch()
        layout.addWidget(scroll)
        return page

    def delete_order(self, order):
        answer = QMessageBox.question(self, "Удаление", "Точно ли удалить заказ")
        if answer == QMessageBox.StandardButton.Yes:
            db.execute("delete from orders where id=%s", (order[0],))
            self.set_status("Заказ удален")
            self.refresh_tab()

    def go_back(self):
        self.close()
        open_login()


def open_login():
    global main_window
    dialog = LoginDialog()
    if dialog.exec():
        main_window = MainWindow(dialog.selected_role, dialog.selected_user)
        main_window.show()
    else:
        QApplication.instance().quit()


if __name__ == "__main__":
    app = QApplication(sys.argv)
    open_login()
    sys.exit(app.exec())
